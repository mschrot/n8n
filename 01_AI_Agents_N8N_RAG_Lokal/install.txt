1.: Ubuntu Server mit Docker + Portainer installieren
https://github.com/mschrot/ubuntu_server


2.: Stack in Portainer implementieren: n8n, ollama service, qdrant (vectorbase)

version: "3.8"

services:
  n8n:
    image: n8nio/n8n:latest
    container_name: n8n
    ports:
      - "5678:5678"
    environment:
      - N8N_SECURE_COOKIE=false
      - N8N_HOST=0.0.0.0
      - N8N_PORT=5678
      - N8N_PROTOCOL=http
      - WEBHOOK_URL=http://IP:5678/
      - GENERIC_TIMEZONE=Europe/Berlin
    volumes:
      - n8n_data:/home/node/.n8n
    restart: always

  ollama:
    image: ollama/ollama
    container_name: ollama
    ports:
      - "11434:11434"
    volumes:
      - ollama_data:/root/.ollama
    restart: always

  qdrant:
    image: qdrant/qdrant
    container_name: qdrant
    ports:
      - "6333:6333"
    volumes:
      - qdrant_data:/qdrant/storage
    restart: always

volumes:
  n8n_data:
  ollama_data:
  qdrant_data:
  

# http://IP:5678/
# Konto einrichten
  
3.: Ubuntu Server Terminal aufmachen mit über SSH: MobaXterm oder Putty (einloggen sudo daten)
